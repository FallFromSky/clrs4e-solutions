\newcommand{\MakeChapter}[4]{%
    \chapter{#1}%
    \foreach \Title [count=\Section] in {#2}{%
        \section{\Title}
        \foreach \Exercise in {1,...,16}{%
            \IfFileExists{#4/sections/\Section/\Exercise}{%
                \exercise
                \import{#4/sections/\Section/}{\Exercise}
            }{}%
        }%
    }%
    \problems
    \foreach \Title [count=\Problem] in {#3}{%
        \problem{\Title}
        \IfFileExists{#4/problems/\Problem/main}{%
            \import{#4/problems/\Problem/}{main}
        }{}%
        \foreach \SubProblem in {a,...,j}{%
            \IfFileExists{#4/problems/\Problem/\SubProblem}{%
                \subproblem
                \import{#4/problems/\Problem/}{\SubProblem}
            }{}%
        }%
    }%
}
