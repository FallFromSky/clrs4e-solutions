\begin{liftalign*}
    \sum_{k=0}^\infty\frac{k-1}{2^k} &= \sum_{k=0}^\infty\frac{k}{2^k}-\sum_{k=0}^\infty\frac{1}{2^k} \\
    &= \sum_{k=0}^\infty k(1/2)^k-\sum_{k=0}^\infty(1/2)^k \\
    &= \frac{1/2}{(1-1/2)^2}-\frac{1}{1-1/2} && \text{(by equations (A.11) and (A.7))} \\
    &= 0
\end{liftalign*}
